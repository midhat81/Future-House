    const body = await req.json();
    contents = body.contents;

      status: 400,
      headers: { "Content-Type": "application/json" },
    });
  }

  const apiKey = Deno.env.get("INTEGRATIONS_API_KEY");
  if (!apiKey) {
    return new Response(JSON.stringify({ error: "Server configuration error" }), {
      status: 500,
      headers: { "Content-Type": "application/json" },
    });

  const upstream = await fetch(
    "https://app-ccr2d5nt3401-api-zYkZzKQJrBdL.gateway.appmedo.com/image-generation/submit",
    {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "X-Gateway-Authorization": `Bearer ${apiKey}`,
      },
      body: JSON.stringify({ contents }),
    }
  );

  if (upstream.status === 429 || upstream.status === 402) {
    const errText = await upstream.text();
    return new Response(errText, {
      status: upstream.status,
      headers: { "Content-Type": "application/json" },
    });
  }

  if (!upstream.ok) {
    return new Response(

          task.status = 'failed';
          task.error = 'Task timed out';
          completedViews.add(task.view);
        }
      } catch (err) {
        console.error(`Error polling ${task.view}:`, err);
      }
    }

    onUpdate([...currentTasks]);
  }

  // Mark remaining pending tasks as failed
  for (const task of currentTasks) {
    if (task.status === 'pending') {
      task.status = 'failed';
      task.error = 'Generation timed out';
    }
  }

  onUpdate([...currentTasks]);
  return currentTasks;
}

        next.delete(id);

      const next = new Set(prev);
      if (next.has(id)) {
        next.delete(id);
      } else {
        if (next.size >= 4) {
          toast.info('You can compare up to 4 designs at a time');
